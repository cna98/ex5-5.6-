import java.util.Scanner;
public class Ex5 {
	
	public static int binarySearch(int[] array , int min , int max , int x)
	{
		if(max >=min)
		{
			int mid = (max + min + (max % 2)) /2;
			
			if(array[mid] == x)
				return x;
			else if(array[mid] > x)
				return binarySearch(array, min, mid-1, x);
			else if(array[mid] < x)
				return binarySearch(array, mid +1, max, x);
			else
				return 1;
			
		}
		else
			return 1;
		
	}
